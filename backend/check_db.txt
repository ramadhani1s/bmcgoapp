package checkdb

import (
	"context"
	"fmt"
	"log"

	"github.com/jackc/pgx/v5"
)

func main() {
	connStr := "postgres://postgres:ramadhani12@localhost:5432/bmcgo_db?sslmode=disable"
	conn, err := pgx.Connect(context.Background(), connStr)
	if err != nil {
		log.Fatal(err)
	}
	defer conn.Close(context.Background())

	_, err = conn.Exec(context.Background(), "UPDATE soal_latihan SET materi_id = 5 WHERE materi_id IS NULL")
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println("Successfully updated orphaned soal_latihan to materi_id = 5")
}
